Analyseur de Données du Marché Freelance

Cet outil interactif utilise Python pour analyser les données de Malt.fr. Entrez un métier et cliquez sur "Analyser" pour voir le tarif journalier et les compétences clés.

packages = ["beautifulsoup4"] import asyncio # === CORRECTION DES IMPORTS === from js import document # On utilise l'objet 'document' global de JavaScript from pyodide.ffi import create_proxy # ============================== from pyodide.http import pyfetch from bs4 import BeautifulSoup import re from collections import Counter def sanitize_html(text): if not isinstance(text, str): text = str(text) return text.replace("&", "&").replace("<", "<").replace(">", ">") # === CORRECTION DE LA SÉLECTION DES ÉLÉMENTS === status_div = document.getElementById("status") results_div = document.getElementById("results") job_input = document.getElementById("job-title") run_button = document.getElementById("run-button") # =============================================== async def main(*args, **kwargs): job_title = sanitize_html(job_input.value) if not job_title: status_div.innerText = "Veuillez entrer un métier." return run_button.disabled = True status_div.innerText = "Initialisation..." results_div.innerHTML = "" await asyncio.sleep(0.1) try: search_query = job_title.replace(" ", "+") malt_url = "https://www.malt.fr/s?q={}".format(search_query) proxy_url = "https://api.allorigins.win/raw?url={}".format(malt_url) status_div.innerText = "Connexion sécurisée au proxy..." response = await pyfetch(url=proxy_url) if response.status != 200: raise ConnectionError("Le proxy a retourné une erreur {}".format(response.status)) status_div.innerText = "Analyse du HTML..." html_content = await response.string() soup = BeautifulSoup(html_content, "html.parser") status_div.innerText = "Extraction et nettoyage des données..." prices = [] price_elements = soup.select('strong[data-testid="price"]') for el in price_elements: price_text = re.search(r'(\d+)', el.text) if price_text: prices.append(int(price_text.group(1))) all_skills = [] skill_elements = soup.select('div[data-testid*="expertise-pills-"]') for el in skill_elements: all_skills.append(sanitize_html(el.text.strip())) status_div.innerText = "Analyse terminée." output_html = "

Résultats pour '{}'

".format(job_title) if prices: average_tjm = sum(prices) / len(prices) output_html += "

Tarif Journalier Moyen : {:.0f} €

".format(average_tjm) else: output_html += "

Aucun tarif trouvé.

" if all_skills: top_skills = Counter(all_skills).most_common(5) output_html += "

Top 5 des compétences :

    " for skill, count in top_skills: output_html += "
  • {}
    • ".format(skill) output_html += "
" else: output_html += "

Aucune compétence trouvée.

" results_div.innerHTML = output_html except Exception as e: status_div.innerText = "Une erreur est survenue." results_div.innerHTML = "

Erreur : Problème de connexion ou de parsing.

" finally: run_button.disabled = False # === CORRECTION DE LA LOGIQUE D'ÉVÉNEMENT === main_proxy = create_proxy(main) run_button.addEventListener("click", main_proxy) # ============================================
